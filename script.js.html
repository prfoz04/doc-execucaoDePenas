        /**
         * formata os dados para a função de geração de gráficos
         * @param {object} dados json recebido via API
         * @param {string} planilha deve ser 'PSC' ou 'PP'
         * @param {string} categoria deve ser uma chave secundária do objeto json
         * @param {string} filtro filtro opcional
         */
        function filtraDados(dados, planilha, categoria, filtro) {
            let retorno = {};
            retorno['labels'] = Object.keys(dados[planilha][categoria]);
            if (filtro)
                retorno['labels'] = retorno['labels'].filter(filtro);
            retorno['dados'] = [];
            for (let key of retorno['labels'])
                retorno['dados'].push(dados[planilha][categoria][key]);  
            retorno['labels'] = retorno['labels'].map(str => str.replace(/_/g, " "));
            let indexEANPP = retorno['labels'].indexOf('CP - EANPP');
            if (indexEANPP !== -1) {
                let indexCP = retorno['labels'].indexOf('CP');
                retorno['dados'][indexCP] += retorno['dados'][indexEANPP];
                retorno['dados'].splice(indexEANPP, 1);
                retorno['labels'].splice(indexEANPP, 1);
            }
            return retorno
        }

        /**
         * utiliza a biblioteca chart para gerar um grafico baseado nos parametros 
         * @param {string} id titulo do grafico
         * @param {string[]} colunas vetor de strings com os nomes de cada coluna ou setor
         * @param {number[]} dados dados associados a cada label 
         * @param {string} medida unidade de medida utilizada
         * @param {string} tipo gráfico de barras = 'bar', gráfico de setores = 'pie'
         */
        function geradorGraficos(id, colunas, dados, medida, tipo) {
            const ctx = document.getElementById(id).getContext('2d');
            new Chart(ctx, {
                type: tipo,
                data: {
                    labels: colunas,
                    datasets: [{
                        label: medida,
                        data: dados,
                        backgroundColor: cores, 
                        borderColor: 'grey',
                        borderWidth: 1
                    }]
                },
                options: {
                    responsive: true,
                    plugins: {
                        legend: {
                            display: false
                        },
                        datalabels: {
                            color: '#333333', // Cor do texto dentro do setor
                            font: {
                                weight: 'bold',
                                size: 14
                            },
                            // Função para calcular e exibir a porcentagem
                            formatter: (value, ctx) => {
                                let soma = 0;
                                let dataArr = ctx.chart.data.datasets[0].data;
                                dataArr.map(data => {
                                    soma += data;
                                });
                                let percentage = (value * 100 / soma).toFixed(0) + "%";
                                if ((value * 100) / soma < 7)
                                    return '';
                                return percentage;
                            }
                        }
                    },
                    scales: {
                        y: {
                            beginAtZero: true,
                            ticks : {
                                display: false
                            }
                        },
                        x: {
                            ticks: {
                                display: false
                            }
                        }
                    }
                }
            })
            inserirLegendas(id, colunas);
            inserirTotal(id, dados);
        }

        /**
         * coloca a entrada no formato #.###,##
         * @param {string} string
         */
        function formataMoeda(string) {
            let valor = string.toString();
            valor = valor.substring(0, valor.indexOf('.') + 3).replace('.', ',')
            valor = valor.split("").reverse()
            let retorno =  '';
            retorno += valor[0] + valor[1] + valor[2];
            for (let i = 3, j = 0; i < valor.length; i++, j++) {
                if (j % 3 == 0 && j != 0)
                    retorno += '.';
                retorno += valor[i];
            }
            return retorno.split("").reverse().join("");
        }
